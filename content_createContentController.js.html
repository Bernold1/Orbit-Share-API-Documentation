    const mongoose = require("mongoose");
const ContentSchema = require("./content");
const bodyParser = require("body-parser");

const errorLogging = require("../errorHandling/errorLogging");

/**
 * gets the content form text from the request parameters, and the file from the request body.
 * Turns the file into a buffer, creates a mongoose Content object based passed data in the request and persists the object.
 * @param   {Object}    req            HTTP Request
 * @param   {Object}    res            HTTP Response
 * @return  {Promise}
 * @async
 * @throws {res.status(400)} If error occurs when trying to persist the document.
 */
const createContent = async (req, res) => {
  let buffer = Buffer.from(req.body);

  const Content = mongoose.model("Content", ContentSchema);

  let uploadedContent = new Content({
    title: req.query.title,
    author: req.query.author,
    description: req.query.description,
    contentData: {
      fileData: buffer,
    },
    access: req.query.access,
    tags: JSON.parse(decodeURIComponent(req.query.tags)),
  });
  console.log("This should be the content: " + uploadedContent);
  try {
    const content = await uploadedContent.save();
    res.status(201).json({ id: content._id });
  } catch (error) {
    errorLogging.handleError(error);
    res.status(400).end();
  }
};

module.exports.createContent = createContent;
