    const mongoose = require("mongoose");
const uniqueValidator = require("mongoose-unique-validator");
const Schema = mongoose.Schema;

/**
 * Mongoose User model
 */
const UserSchema = new Schema({
  username: {
    type: String,
    required: true,
    unique: true,
  },
  email: {
    type: String,
    required: true,
  },
  password: {
    type: String,
    required: true,
  },
  firstName: {
    type: String,
  },
  lastName: {
    type: String,
  },
  description: {
    type: String,
  },
});

// run validator on unique tags - https://mongoosejs.com/docs/validation.html#the-unique-option-is-not-a-validator
UserSchema.plugin(uniqueValidator);

const User = mongoose.model("User", UserSchema);

module.exports.UserSchema = UserSchema;
module.exports = User;
