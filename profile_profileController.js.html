    //Node.js modules and middleware
const mongoose = require("mongoose");

//Models
const User = require("../user/user");
const Collection = require("../collection/collection");
const ContentSchema = require("../content/content");

//Controllers
const errorLogging = require("../errorHandling/errorLogging");

/**
 * Gets specified username based on passed id in request query.
 * @param   {Object}    req            HTTP Request
 * @param   {Object}    res            HTTP Response
 * @return  {Promise}
 * @async
 * @throws {res.status(404)} If error occurs when trying to find user
 */
const getUsername = async (req, res) => {
  const userId = req.query.userId;
  try {
    const userInfo = await User.findById(
      { _id: userId },
      { email: 0, password: 0, firstName: 0, lastName: 0, description: 0 }
    );
    res.status(200).json(userInfo);
  } catch (error) {
    errorLogging.handleError(error);
    res.status(400).end();
  }
};

/**
 * Gets all content based on passed id in request query for a specified user.
 * @param   {Object}    req            HTTP Request
 * @param   {Object}    res            HTTP Response
 * @return  {Promise}
 * @async
 * @throws {res.status(400)} If error occurs when trying to find document/document is not found
 */
const getUserContent = async (req, res) => {
  const userId = req.query.userId;
  const Content = mongoose.model("Content", ContentSchema);
  try {
    let userContent = await Content.find(
      { author: userId },
      { contentData: 0 }
    );
    res.status(200).json({ userContent });
  } catch (error) {
    errorLogging.handleError(error);
    res.status(400).end();
  }
};

/**
 * Gets all collections based on passed id in request query for a specified user.
 * @param   {Object}    req            HTTP Request
 * @param   {Object}    res            HTTP Response
 * @return  {Promise}
 * @async
 * @throws {res.status(400)} If error occurs when trying to find document/document is not found
 */
const getUserCollections = async (req, res) => {
  const userId = req.query.userId;

  try {
    let userCollection = await Collection.find(
      { userId: userId },
      { "content.contentData": 0 }
    );
    res.status(200).json({ userCollection });
  } catch (error) {
    errorLogging.handleError(error);
    res.status(400).end();
  }
};

module.exports.getUsername = getUsername;
module.exports.getUserContent = getUserContent;
module.exports.getUserCollections = getUserCollections;
